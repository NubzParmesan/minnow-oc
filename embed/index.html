JavaScript (fetch tuna.json and render)

This pulls the canon phrases from tuna.json and renders the first 3.

<div id="tuna"></div>
<script>
(async () => {
  const res = await fetch('https://nubzparmesan.github.io/minnow-oc/tuna.json', { cache: 'force-cache' });
  const data = await res.json();

  const lines = (data.canonPhrases || []).slice(0, 3);
  const el = document.getElementById('tuna');

  el.style.fontFamily = 'monospace';
  el.style.lineHeight = '1.5';
  el.innerHTML = '';

  const a = document.createElement('a');
  a.href = 'https://nubzparmesan.github.io/minnow-oc/start/';
  a.textContent = lines[0] || 'tuna is the constraint.';
  a.style.color = '#7eb8f7';
  a.style.textDecoration = 'none';
  el.appendChild(a);
  el.appendChild(document.createElement('br'));

  (lines.slice(1).length ? lines.slice(1) : ['memory is sacred.', 'curation is identity.'])
    .forEach((t) => {
      el.appendChild(document.createTextNode(t));
      el.appendChild(document.createElement('br'));
    });
})();
</script>
